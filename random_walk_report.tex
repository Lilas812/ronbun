\section{ランダムウォークについて}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\subsection{ランダムウォークの定義}

ランダムウォーク（random walk）とは，「酔っ払いの歩き方」に例えられる確率的な移動モデルである．
一次元の整数格子 $\mathbb{Z}$ 上を歩く粒子（ウォーカー）は，各時刻ごとにコインを投げて
右か左かを決め，その方向へ 1 歩進む．

\begin{definition}[ランダムウォーク]\label{def:rw}
確率空間 $(\Omega,\mathcal{F},P)$ 上の独立同分布な確率変数列 $T_1, T_2, \ldots$ を
\[
  P(T_k = +1) = p, \quad P(T_k = -1) = q = 1-p
  \qquad (0 < p < 1)
\]
によって定める．このとき，$n$ 時刻後の位置
\[
  S_n = T_1 + T_2 + \cdots + T_n, \quad S_0 = 0
\]
を（一次元単純）\textbf{ランダムウォーク}という．
$p$ を右進確率，$q$ を左進確率と呼ぶ．
\end{definition}

\begin{remark}
$p = q = 1/2$ の場合を\textbf{対称ランダムウォーク}，$p \neq q$ の場合を
\textbf{非対称ランダムウォーク}という．
\end{remark}

%---------------------------------------------------------
\subsection{ランダムウォークの確率分布}
%---------------------------------------------------------

\begin{proposition}[確率分布]\label{prop:rw_dist}
$(n+x)/2$ が非負整数のとき，
\[
  P(S_n = x)
  = \binom{n}{\dfrac{n+x}{2}} \, p^{(n+x)/2} \, q^{(n-x)/2}.
\]
それ以外のとき $P(S_n = x) = 0$．
\end{proposition}

\begin{proof}[証明の概略]
$n$ 回の移動のうち右へ進む回数を $r$，左へ進む回数を $\ell$ とすると，
$r + \ell = n$ かつ $r - \ell = x$ より $r = (n+x)/2$，$\ell = (n-x)/2$．
$\binom{n}{r}$ 通りの順序それぞれに確率 $p^r q^{\ell}$ が対応する．
\end{proof}

\subsubsection*{$n = 1, 2, 3$ の具体例}

\begin{example}[$n=1$]
\[
  P(S_1 = +1) = p, \qquad P(S_1 = -1) = q.
\]
\end{example}

\begin{example}[$n=2$]
\[
  P(S_2 = +2) = p^2, \qquad
  P(S_2 = 0) = 2pq, \qquad
  P(S_2 = -2) = q^2.
\]
\end{example}

\begin{example}[$n=3$]
\[
  P(S_3 = +3) = p^3, \quad
  P(S_3 = +1) = 3p^2 q, \quad
  P(S_3 = -1) = 3p q^2, \quad
  P(S_3 = -3) = q^3.
\]
\end{example}

\subsubsection*{平均と分散}

\begin{proposition}[モーメント]\label{prop:rw_moment}
$S_0 = 0$ とするとき，
\begin{align}
  \mathbb{E}[S_n]     &= (p - q)\,n, \label{eq:rw_mean} \\
  \mathrm{Var}(S_n)   &= 4pq\,n.     \label{eq:rw_var}
\end{align}
\end{proposition}

\begin{proof}[証明の概略]
各 $T_k$ が独立であるから，期待値の線型性より
$\mathbb{E}[S_n] = n\,\mathbb{E}[T_1] = n(p - q)$．
また $\mathrm{Var}(T_1) = \mathbb{E}[T_1^2] - (\mathbb{E}[T_1])^2 = 1 - (p-q)^2 = 4pq$
なので $\mathrm{Var}(S_n) = 4pqn$．
\end{proof}

\begin{remark}
$p = 1/2$ のとき $\mathbb{E}[S_n] = 0$，$\mathrm{Var}(S_n) = n$．
ウォーカーの典型的な広がりは $\sqrt{n}$ のオーダーであり，これを\textbf{拡散的挙動}という．
\end{remark}

\section{相関付きランダムウォークについて}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\subsection{相関付きランダムウォークの定義}

通常のランダムウォークでは各ステップが独立であるが，
\textbf{相関付きランダムウォーク}（correlated random walk）では
直前の移動方向が次のステップに影響する．

\begin{definition}[相関付きランダムウォーク]
パラメータ $p \in (0,1)$，$q = 1-p$ とする．
ウォーカーの各ステップは次の遷移確率に従う：
\[
  P(\text{右} \to \text{右}) = p, \quad
  P(\text{右} \to \text{左}) = q, \quad
  P(\text{左} \to \text{左}) = p, \quad
  P(\text{左} \to \text{右}) = q.
\]
すなわち，直前と\textbf{同じ方向}に進む確率が $p$，
\textbf{逆方向}に進む確率が $q$ である．
このとき定まる位置過程 $(\tilde{S}_n)_{n \geq 0}$ を
\textbf{相関付きランダムウォーク}という．
\end{definition}

\begin{remark}
$p > 1/2$ のとき同方向への持続性が強く（持続的ウォーク），
$p < 1/2$ のとき反転しやすくなり（反持続的ウォーク），
$p = 1/2$ のとき通常のランダムウォークに一致する．
\end{remark}

\subsubsection*{行列による定式化}

内部状態（向き）を $\{-, +\}$（左・右）とする 2 成分ベクトルで記述すると，
相関付きランダムウォークは $2 \times 2$ 行列で表現できる．

\begin{definition}[推移行列]\label{def:crw_matrix}
右移動行列 $P$（位置 $x-1$ から $x$ への右向き寄与）と
左移動行列 $Q$（位置 $x+1$ から $x$ への左向き寄与）を
\[
  P = \begin{pmatrix} 0 & 0 \\ q & p \end{pmatrix},
  \qquad
  Q = \begin{pmatrix} p & q \\ 0 & 0 \end{pmatrix}
\]
と定め，\textbf{推移行列}を
\[
  A = P + Q = \begin{pmatrix} p & q \\ q & p \end{pmatrix}
\]
とする．
\end{definition}

\begin{definition}[確率ベクトル]\label{def:crw_vec}
時刻 $n$，位置 $x$ における確率ベクトルを
\[
  \Psi_n(x) = \begin{pmatrix} \Psi_n^{-}(x) \\ \Psi_n^{+}(x) \end{pmatrix}
\]
で表す．$\Psi_n^{-}(x)$ は時刻 $n$ に左向きで位置 $x$ にいる確率，
$\Psi_n^{+}(x)$ は右向きで位置 $x$ にいる確率である．
初期状態は $\Psi_0(0) = \begin{pmatrix} \alpha \\ \beta \end{pmatrix}$
（$\alpha + \beta = 1$，$\alpha,\beta \geq 0$）とする．
\end{definition}

時刻発展方程式は
\begin{equation}
  \Psi_{n+1}(x) = P\,\Psi_n(x-1) + Q\,\Psi_n(x+1)
  \label{eq:crw_evolution}
\end{equation}
で与えられる．$P$ は位置 $x-1$ から右向きに $x$ へ移動した寄与（$\Psi^+$ 成分を更新），
$Q$ は位置 $x+1$ から左向きに $x$ へ移動した寄与（$\Psi^-$ 成分を更新）に対応する．

%---------------------------------------------------------
\subsection{相関付きランダムウォークの確率分布}
%---------------------------------------------------------

時刻 $n$ に位置 $x$ にいる確率は，左向き・右向きの寄与の和として
\begin{equation}
  P(\tilde{S}_n = x) = \Psi_n^{-}(x) + \Psi_n^{+}(x)
  \label{eq:crw_prob}
\end{equation}
で与えられる．

\begin{proposition}[確率の正規化]
任意の $n \geq 0$ に対して
$\displaystyle\sum_{x \in \mathbb{Z}} P(\tilde{S}_n = x) = 1$．
\end{proposition}

\subsubsection*{$n = 1, 2, 3$ の具体例}

以下では初期状態を $\Psi_0(0) = (\alpha,\,\beta)^\top$（$\alpha + \beta = 1$）とする．

\begin{example}[$n=1$]
\begin{align*}
  P(\tilde{S}_1 = +1) &= p\beta + q\alpha, \\
  P(\tilde{S}_1 = -1) &= p\alpha + q\beta.
\end{align*}
\end{example}

\begin{example}[$n=2$]
\begin{align*}
  P(\tilde{S}_2 = +2) &= p\,(p\beta + q\alpha), \\
  P(\tilde{S}_2 =  0) &= q\,(p\beta + q\alpha) + q\,(p\alpha + q\beta) = q, \\
  P(\tilde{S}_2 = -2) &= p\,(p\alpha + q\beta).
\end{align*}
$P(\tilde{S}_2 = 0) = q$ は $\alpha + \beta = 1$ から直接導かれる．
\end{example}

\begin{example}[$n=3$]
\begin{align*}
  P(\tilde{S}_3 = +3) &= p^2\,(p\beta + q\alpha), \\
  P(\tilde{S}_3 = +1) &= q^2\,(p\beta + q\alpha) + pq, \\
  P(\tilde{S}_3 = -1) &= q^2\,(p\alpha + q\beta) + pq, \\
  P(\tilde{S}_3 = -3) &= p^2\,(p\alpha + q\beta).
\end{align*}
\end{example}

\subsubsection*{平均}

各ステップ $k$ での期待変位は，直前の向きの期待値に $(p-q)$ を掛けて得られる．
$\mathbb{E}[\text{dir}_k] = (p-q)^k(\beta-\alpha)$（$\text{dir}_k \in \{+1,-1\}$）であるから，

\begin{proposition}[1次モーメント]\label{prop:crw_mean}
$\Psi_0(0) = (\alpha,\,\beta)^\top$ とするとき，
\begin{equation}
  \mathbb{E}[\tilde{S}_n]
  = (\beta - \alpha)\sum_{k=1}^{n}(p-q)^k
  = \frac{(p-q)(\beta-\alpha)\bigl[1-(p-q)^n\bigr]}{2q}
  \qquad (p \neq q). \label{eq:crw_mean}
\end{equation}
$p = q = 1/2$ のとき $\mathbb{E}[\tilde{S}_n] = 0$，$\mathrm{Var}(\tilde{S}_n) = n$．
\end{proposition}

\begin{remark}
$|p-q| < 1$ であるから $(p-q)^n \to 0$（$n \to \infty$）．
したがって $\mathbb{E}[\tilde{S}_n]$ は $n \to \infty$ で有限値
$\dfrac{(p-q)(\beta-\alpha)}{2q}$ に収束する（$p \neq q$）．
分散は $n$ に比例して増大し（拡散的挙動），$p = 1/2$ のときは
通常のランダムウォークと同じ $\mathrm{Var}(\tilde{S}_n) = n$ が成り立つ．
\end{remark}
