\section{ランダムウォークについて}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

本章では，1 次元ランダムウォークの基本的な性質について述べる。
まず，ランダムウォークの定義とともに，時刻 $n$ において位置がどのような確率で決まるかを概観する。
次に，確率分布 $P(S_n = x)$ を具体的に導出し，最後に極限分布について説明する。

%---------------------------------------------------------
\subsection{ランダムウォークの定義}
%---------------------------------------------------------

確率変数列 $\{T_n\}_{n \geq 1}$ を考え，各 $T_n$ は独立同分布で
\begin{equation}
  P(T_n = -1) = p, \quad P(T_n = 1) = q, \qquad (p + q = 1) \label{eq:rv}
\end{equation}
を満たすとする。
このとき，時刻 $n$ におけるランダムウォークの位置 $S_n$ を
\begin{equation}
  S_n = \sum_{k=1}^{n} T_k \label{eq:Sn}
\end{equation}
と定義する。ただし初期位置は $S_0 = 0$ とする。

この定義から，$S_n$ は $n$ 回の独立な試行の結果として定まる確率変数である。
実際，左方向（$T_k = -1$）への移動回数を $\ell$，右方向（$T_k = +1$）への移動回数を $m$ とすると，
$\ell + m = n$ が成り立ち，このとき
\begin{equation}
  x = m - \ell \label{eq:pos}
\end{equation}
である。

%---------------------------------------------------------
\subsection{ランダムウォークの確率分布}
%---------------------------------------------------------

次に，時刻 $n$ における位置 $S_n$ の確率分布 $P(S_n = x)$ を考える。
$S_n = x$ が成立するためには，左方向への移動回数 $\ell$ と右方向への移動回数 $m$ が
\begin{equation}
  m - \ell = x, \qquad \ell + m = n \label{eq:lm}
\end{equation}
の関係式が成立する。図\,\ref{fig:path} は $S_7 = 1$ となる 1 つのパスを示した。

$T_n$ が独立であることから，左方向へ $\ell$ 回，右方向へ $m$ 回移動する確率は
$\binom{n}{\ell} p^\ell q^m$ で与えられる。
したがって，事象「$S_n = x$」の確率は
\begin{equation}
  P(S_n = x) = \binom{n}{\ell} p^\ell q^m
  = \frac{n!}{\ell!\,(n-\ell)!}\,p^\ell q^m
  \label{eq:dist}
\end{equation}
のようになる。ただし $\ell = (n-x)/2$，$m = (n+x)/2$ とする
（$(n+x)/2$ が非負整数でないとき $P(S_n=x)=0$）。

例えば，時刻 $n = 1$ のとき，
\begin{equation}
  P(S_1 = -1) = p, \qquad P(S_1 = 1) = q \label{eq:n1}
\end{equation}
である。同様にして時刻 $n = 2$ では，
\begin{equation}
  P(S_2 = -2) = p^2, \qquad P(S_2 = 0) = 2pq, \qquad P(S_2 = 2) = q^2
  \label{eq:n2}
\end{equation}
となる。さらに，$n = 3$ では，
\begin{align}
  P(S_3 = -3) = \binom{3}{3} p^3 q^0 = p^3, &\qquad
  P(S_3 = -1) = \binom{3}{2} p^2 q^1 = 3p^2 q, \label{eq:n3a}\\
  P(S_3 = 1)  = \binom{3}{1} p^1 q^2 = 3pq^2, &\qquad
  P(S_3 = 3)  = \binom{3}{0} p^0 q^3 = q^3   \label{eq:n3b}
\end{align}
のようになる。

また，$p = q = 1/2$ の対称なランダムウォークの $P(S_n = x)$ を図示すると
図\,\ref{fig:barchart} のようになる。
このとき，時刻 $n$ が偶数のときはランダムウォークの取りえる場所も偶数で，奇数の時刻では場所も奇数となり，\textbf{偶奇性}が成り立つ。

%---------------------------------------------------------
\subsection{ランダムウォークの極限分布}
%---------------------------------------------------------

最後に，$n$ を大きくしたときの $S_n$ の分布について考える。
$\mathbb{E}[S_n]$ を平均，$\mathrm{Var}[S_n]$ を分散とすると，中心極限定理より，
\begin{equation}
  \frac{S_n - \mathbb{E}[S_n]}{\sqrt{\mathrm{Var}[S_n]}} \label{eq:clt_form}
\end{equation}
は $n \to \infty$ のとき標準正規分布に弱収束する。
一般の $p, q$ に対して
\begin{equation}
  \mathbb{E}[S_n] = (q - p)\,n, \qquad \mathrm{Var}[S_n] = 4pq\,n. \label{eq:moments}
\end{equation}

特に $p = q = 1/2$ の場合には
\begin{equation}
  \mathbb{E}[S_n] = 0, \qquad \mathrm{Var}[S_n] = n \label{eq:moments_sym}
\end{equation}
であるから，
\begin{equation}
  \frac{S_n}{\sqrt{n}} \label{eq:scaled}
\end{equation}
は図\,\ref{fig:normal} のような標準正規分布に収束する。
したがって，任意の実数 $a < b$ に対して
\begin{equation}
  \lim_{n \to \infty} P\!\left(a \leq \frac{S_n}{\sqrt{n}} \leq b\right)
  = \int_a^b \frac{1}{\sqrt{2\pi}}\,e^{-x^2/2}\,dx \label{eq:clt}
\end{equation}
が成り立つ。

\section{相関付きランダムウォークについて}

\subsection{相関付きランダムウォークの定義}

通常のランダムウォークでは各ステップが独立であるが，
\textbf{相関付きランダムウォーク}（correlated random walk）では
直前の移動方向が次のステップに影響する。

\begin{definition}[相関付きランダムウォーク]
パラメータ $a \in (0,1)$，$b = 1-a$ とする。
ウォーカーの各ステップは次の遷移確率に従う：
\[
  P(\text{左} \to \text{左}) = a, \quad
  P(\text{左} \to \text{右}) = b, \quad
  P(\text{右} \to \text{右}) = a, \quad
  P(\text{右} \to \text{左}) = b.
\]
すなわち，直前と\textbf{同じ方向}に進む確率が $a$，
\textbf{逆方向}に進む確率が $b$ である。
このとき定まる位置過程 $(\tilde{S}_n)_{n \geq 0}$ を
\textbf{相関付きランダムウォーク}という。
\end{definition}

\begin{remark}
$a > 1/2$ のとき同方向への持続性が強く（持続的ウォーク），
$a < 1/2$ のとき反転しやすくなり（反持続的ウォーク），
$a = 1/2$ のとき通常のランダムウォークに一致する。
\end{remark}

\subsubsection*{行列による定式化}

内部状態（向き）を $\{-, +\}$（左・右）とする 2 成分ベクトルで記述すると，
相関付きランダムウォークは $2 \times 2$ 行列で表現できる。

\begin{definition}[推移行列]
右移動行列 $P_R$（位置 $x-1$ から $x$ への右向き寄与）と
左移動行列 $P_L$（位置 $x+1$ から $x$ への左向き寄与）を
\[
  P_R = \begin{pmatrix} 0 & 0 \\ b & a \end{pmatrix},
  \qquad
  P_L = \begin{pmatrix} a & b \\ 0 & 0 \end{pmatrix}
\]
と定め，\textbf{推移行列}を
\[
  A = P_R + P_L = \begin{pmatrix} a & b \\ b & a \end{pmatrix}
\]
とする。
\end{definition}

\begin{definition}[確率ベクトル]
時刻 $n$，位置 $x$ における確率ベクトルを
\[
  \Psi_n(x) = \begin{pmatrix} \Psi_n^{-}(x) \\ \Psi_n^{+}(x) \end{pmatrix}
\]
で表す。$\Psi_n^{-}(x)$ は時刻 $n$ に左向きで位置 $x$ にいる確率，
$\Psi_n^{+}(x)$ は右向きで位置 $x$ にいる確率である。
初期状態は $\Psi_0(0) = \begin{pmatrix} \alpha \\ \beta \end{pmatrix}$
（$\alpha + \beta = 1$，$\alpha,\beta \geq 0$）とする。
\end{definition}

時刻発展方程式は
\begin{equation}
  \Psi_{n+1}(x) = P_R\,\Psi_n(x-1) + P_L\,\Psi_n(x+1)
  \label{eq:crw_evolution}
\end{equation}
で与えられる。$P_R$ は位置 $x-1$ から右向きに $x$ へ移動した寄与（$\Psi^+$ 成分を更新），
$P_L$ は位置 $x+1$ から左向きに $x$ へ移動した寄与（$\Psi^-$ 成分を更新）に対応する。

%---------------------------------------------------------
\subsection{相関付きランダムウォークの確率分布}
%---------------------------------------------------------

時刻 $n$ に位置 $x$ にいる確率は，左向き・右向きの寄与の和として
\begin{equation}
  P(\tilde{S}_n = x) = \Psi_n^{-}(x) + \Psi_n^{+}(x)
  \label{eq:crw_prob}
\end{equation}
で与えられる。

\begin{proposition}[確率の正規化]
任意の $n \geq 0$ に対して
$\displaystyle\sum_{x \in \mathbb{Z}} P(\tilde{S}_n = x) = 1$。
\end{proposition}

\subsubsection*{$n = 1, 2, 3$ の具体例}

以下では初期状態を $\Psi_0(0) = (\alpha,\,\beta)^\top$（$\alpha + \beta = 1$）とする。

\medskip
\noindent\textbf{$n=1$ のとき：}
\begin{align*}
  P(\tilde{S}_1 = +1) &= a\beta + b\alpha, \\
  P(\tilde{S}_1 = -1) &= a\alpha + b\beta.
\end{align*}

\medskip
\noindent\textbf{$n=2$ のとき：}
\begin{align*}
  P(\tilde{S}_2 = +2) &= a\,(a\beta + b\alpha), \\
  P(\tilde{S}_2 =  0) &= b\,(a\beta + b\alpha) + b\,(a\alpha + b\beta) = b, \\
  P(\tilde{S}_2 = -2) &= a\,(a\alpha + b\beta).
\end{align*}
$P(\tilde{S}_2 = 0) = b$ は $\alpha + \beta = 1$ から直接導かれる。

\medskip
\noindent\textbf{$n=3$ のとき：}
\begin{align*}
  P(\tilde{S}_3 = +3) &= a^2\,(a\beta + b\alpha), \\
  P(\tilde{S}_3 = +1) &= b^2\,(a\beta + b\alpha) + ab, \\
  P(\tilde{S}_3 = -1) &= b^2\,(a\alpha + b\beta) + ab, \\
  P(\tilde{S}_3 = -3) &= a^2\,(a\alpha + b\beta).
\end{align*}

\subsubsection*{平均}

\begin{proposition}[1次モーメント]
$\Psi_0(0) = (\alpha,\,\beta)^\top$ とするとき，
\begin{equation}
  \mathbb{E}[\tilde{S}_n]
  = (\beta - \alpha)\sum_{k=1}^{n}(a-b)^k
  = \frac{(a-b)(\beta-\alpha)\bigl[1-(a-b)^n\bigr]}{2b}
  \qquad (a \neq b). \label{eq:crw_mean}
\end{equation}
$a = b = 1/2$ のとき $\mathbb{E}[\tilde{S}_n] = 0$，$\mathrm{Var}(\tilde{S}_n) = n$。
\end{proposition}

\begin{remark}
$|a-b| < 1$ であるから，$n \to \infty$ で $(a-b)^n \to 0$。
よって $\mathbb{E}[\tilde{S}_n]$ は有限値 $\dfrac{(a-b)(\beta-\alpha)}{2b}$ に収束する（$a \neq b$）。
分散は $n$ に比例して増大し（拡散的挙動），$a = 1/2$ のときは
通常のランダムウォークと同様に $\mathrm{Var}(\tilde{S}_n) = n$ が成り立つ。
\end{remark}
