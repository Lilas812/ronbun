\section{ランダムウォークについて}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

本章では，1 次元ランダムウォークの基本的な性質について述べる。
まず，ランダムウォークの定義とともに，時刻 $n$ において位置がどのような確率で決まるかを概観する。
次に，確率分布 $P(S_n = x)$ を具体的に導出し，最後に極限分布について説明する。

%---------------------------------------------------------
\subsection{ランダムウォークの定義}
%---------------------------------------------------------

確率変数列 $\{T_n\}_{n \geq 1}$ を考え，各 $T_n$ は独立同分布で
\begin{equation}
  P(T_n = -1) = p, \quad P(T_n = 1) = q, \qquad (p + q = 1) \label{eq:rv}
\end{equation}
を満たすとする。
このとき，時刻 $n$ におけるランダムウォークの位置 $S_n$ を
\begin{equation}
  S_n = \sum_{k=1}^{n} T_k \label{eq:Sn}
\end{equation}
と定義する。ただし初期位置は $S_0 = 0$ とする。

この定義から，$S_n$ は $n$ 回の独立な試行の結果として定まる確率変数である。
実際，左方向（$T_k = -1$）への移動回数を $\ell$，右方向（$T_k = +1$）への移動回数を $m$ とすると，
$\ell + m = n$ が成り立ち，このとき
\begin{equation}
  x = m - \ell \label{eq:pos}
\end{equation}
である。

%---------------------------------------------------------
\subsection{ランダムウォークの確率分布}
%---------------------------------------------------------

次に，時刻 $n$ における位置 $S_n$ の確率分布 $P(S_n = x)$ を考える。
$S_n = x$ が成立するためには，左方向への移動回数 $\ell$ と右方向への移動回数 $m$ が
\begin{equation}
  m - \ell = x, \qquad \ell + m = n \label{eq:lm}
\end{equation}
の関係式が成立する。図\,\ref{fig:path} は $S_7 = 1$ となる 1 つのパスを示した。

$T_n$ が独立であることから，左方向へ $\ell$ 回，右方向へ $m$ 回移動する確率は
$\binom{n}{\ell} p^\ell q^m$ で与えられる。
したがって，事象「$S_n = x$」の確率は
\begin{equation}
  P(S_n = x) = \binom{n}{\ell} p^\ell q^m
  = \frac{n!}{\ell!\,(n-\ell)!}\,p^\ell q^m
  \label{eq:dist}
\end{equation}
のようになる。ただし $\ell = (n-x)/2$，$m = (n+x)/2$ とする
（$(n+x)/2$ が非負整数でないとき $P(S_n=x)=0$）。

例えば，時刻 $n = 1$ のとき，
\begin{equation}
  P(S_1 = -1) = p, \qquad P(S_1 = 1) = q \label{eq:n1}
\end{equation}
である。同様にして時刻 $n = 2$ では，
\begin{equation}
  P(S_2 = -2) = p^2, \qquad P(S_2 = 0) = 2pq, \qquad P(S_2 = 2) = q^2
  \label{eq:n2}
\end{equation}
となる。さらに，$n = 3$ では，
\begin{align}
  P(S_3 = -3) = \binom{3}{3} p^3 q^0 = p^3, &\qquad
  P(S_3 = -1) = \binom{3}{2} p^2 q^1 = 3p^2 q, \label{eq:n3a}\\
  P(S_3 = 1)  = \binom{3}{1} p^1 q^2 = 3pq^2, &\qquad
  P(S_3 = 3)  = \binom{3}{0} p^0 q^3 = q^3   \label{eq:n3b}
\end{align}
のようになる。

また，$p = q = 1/2$ の対称なランダムウォークの $P(S_n = x)$ を図示すると
図\,\ref{fig:barchart} のようになる。
このとき，時刻 $n$ が偶数のときはランダムウォークの取りえる場所も偶数で，奇数の時刻では場所も奇数となり，\textbf{偶奇性}が成り立つ。

%---------------------------------------------------------
\subsection{ランダムウォークの極限分布}
%---------------------------------------------------------

最後に，$n$ を大きくしたときの $S_n$ の分布について考える。
$\mathbb{E}[S_n]$ を平均，$\mathrm{Var}[S_n]$ を分散とすると，中心極限定理より，
\begin{equation}
  \frac{S_n - \mathbb{E}[S_n]}{\sqrt{\mathrm{Var}[S_n]}} \label{eq:clt_form}
\end{equation}
は $n \to \infty$ のとき標準正規分布に弱収束する。
一般の $p, q$ に対して
\begin{equation}
  \mathbb{E}[S_n] = (q - p)\,n, \qquad \mathrm{Var}[S_n] = 4pq\,n. \label{eq:moments}
\end{equation}

特に $p = q = 1/2$ の場合には
\begin{equation}
  \mathbb{E}[S_n] = 0, \qquad \mathrm{Var}[S_n] = n \label{eq:moments_sym}
\end{equation}
であるから，
\begin{equation}
  \frac{S_n}{\sqrt{n}} \label{eq:scaled}
\end{equation}
は図\,\ref{fig:normal} のような標準正規分布に収束する。
したがって，任意の実数 $a < b$ に対して
\begin{equation}
  \lim_{n \to \infty} P\!\left(a \leq \frac{S_n}{\sqrt{n}} \leq b\right)
  = \int_a^b \frac{1}{\sqrt{2\pi}}\,e^{-x^2/2}\,dx \label{eq:clt}
\end{equation}
が成り立つ。

\section{相関付きランダムウォークについて}

\subsection{相関付きランダムウォークの定義}

\begin{remark}
本節以降，$p, q$ は §1 の左進確率・右進確率とは異なり，
相関付きランダムウォークの持続パラメータを表す（後述）。
\end{remark}

通常のランダムウォークでは各ステップが独立であるが，
\textbf{相関付きランダムウォーク}（correlated random walk, CRW）では
直前の移動方向が次のステップの確率に影響する。

\begin{definition}[相関付きランダムウォーク]
パラメータ $p, q \in (0,1)$ を固定する。
各ステップの条件付き確率を
\begin{align*}
  P(\text{左} \mid \text{直前：左}) &= p, \quad &
  P(\text{右} \mid \text{直前：左}) &= 1-p, \\
  P(\text{左} \mid \text{直前：右}) &= 1-q, \quad &
  P(\text{右} \mid \text{直前：右}) &= q
\end{align*}
で定める。$p$ を左持続確率，$q$ を右持続確率とよぶ。
このとき定まる位置過程 $(\tilde{S}_n)_{n \geq 0}$ を
\textbf{相関付きランダムウォーク}という。
\end{definition}

\begin{remark}[特殊例]
\begin{itemize}
  \item $p = q$：\textbf{時間発展対称な CRW}。左右の持続確率が等しい。
  \item $p + q = 1$：各ステップが独立となり，通常のランダムウォークに帰着する。
  \item $p = q = 1/2$：対称ランダムウォークに帰着する。
\end{itemize}
\end{remark}

\subsubsection*{行列による定式化}

内部状態（向き）を
\[
  L = \begin{pmatrix}1\\0\end{pmatrix} \text{（左向き）}, \qquad
  R = \begin{pmatrix}0\\1\end{pmatrix} \text{（右向き）}
\]
として，相関付きランダムウォークを $2 \times 2$ 行列で定式化する。

\begin{definition}[転置推移確率行列]
\[
  A = \begin{pmatrix} a & b \\ c & d \end{pmatrix}, \qquad
  a + c = b + d = 1, \quad 0 < a,\, d < 1
\]
を\textbf{転置推移確率行列}とよぶ。
$p, q$ との対応は
\[
  a = p, \quad b = 1-q, \quad c = 1-p, \quad d = q
\]
であり，$AL = aL + cR$，$AR = bL + dR$ が成り立つ。
\end{definition}

\begin{definition}[確率ベクトル]
時刻 $n$，位置 $x$ における\textbf{確率ベクトル}を
\[
  \Psi_n(x) = \begin{pmatrix} \Psi_n^L(x) \\ \Psi_n^R(x) \end{pmatrix}
\]
で表す。$\Psi_n^L(x)$（resp.\ $\Psi_n^R(x)$）は時刻 $n$ に
左向き（resp.\ 右向き）で位置 $x$ にある確率である。
正規化条件 $\displaystyle\sum_{x \in \mathbb{Z}} \bigl(\Psi_n^L(x) + \Psi_n^R(x)\bigr) = 1$
を仮定し，時刻 $n$ に位置 $x$ にいる確率を
\[
  P(\tilde{S}_n = x) = \Psi_n^L(x) + \Psi_n^R(x) = \|\Psi_n(x)\|_1
\]
と定義する。
\end{definition}

行列 $A$ を左移動成分と右移動成分に分解する：
\[
  P = \begin{pmatrix} a & b \\ 0 & 0 \end{pmatrix}, \qquad
  Q = \begin{pmatrix} 0 & 0 \\ c & d \end{pmatrix}, \qquad
  A = P + Q.
\]
$P$ は左向きへの寄与（位置 $x+1$ から $x$ への左向き移動），
$Q$ は右向きへの寄与（位置 $x-1$ から $x$ への右向き移動）を担う。
確率ベクトルの時間発展は
\begin{equation}
  \Psi_{n+1}(x) = P\,\Psi_n(x+1) + Q\,\Psi_n(x-1)
  \label{eq:crw_evolution}
\end{equation}
で与えられる。

初期確率ベクトルを
$\varphi = (\alpha,\,\beta)^\top$，$\alpha + \beta = 1$，$\alpha,\beta \geq 0$
とし，初期状態は原点 $\Psi_0(0) = \varphi$ とする。
$\alpha = 0$（resp.\ $\beta = 0$）は右向き（resp.\ 左向き）から出発することに対応する。

%---------------------------------------------------------
\subsection{相関付きランダムウォークの確率分布}
%---------------------------------------------------------

\subsubsection*{パス和 $\Xi_n(l,m)$}

$n$ ステップで左に $l$ 回，右に $m$ 回（$l + m = n$）移動するとき，
$l$ 個の $P$（左移動行列）と $m$ 個の $Q$（右移動行列）の積の全順序づけの和を
$\Xi_n(l, m)$ と定義する。
位置と試行回数の関係は $x = m - l$，$l = \dfrac{n-x}{2}$，$m = \dfrac{n+x}{2}$ である。

$n = 2$ の場合：
\begin{align*}
  \Xi_2(2,0) &= P^2, \quad \Xi_2(0,2) = Q^2, \\
  \Xi_2(1,1) &= QP + PQ.
\end{align*}

$n = 3$ の場合：
\begin{align*}
  \Xi_3(3,0) &= P^3, \quad \Xi_3(0,3) = Q^3, \\
  \Xi_3(2,1) &= QP^2 + PQP + P^2Q, \\
  \Xi_3(1,2) &= PQ^2 + QPQ + Q^2P.
\end{align*}

\begin{remark}
通常のランダムウォークでは $\Xi_2(1,1)$ に対応する量はスカラー $2pq$ であるが，
相関付きランダムウォークでは $P$ と $Q$ が行列で\textbf{非可換}であるため，
$\Xi_2(1,1) = QP + PQ \neq 2PQ$ が一般に成り立つ。
\end{remark}

\begin{proposition}[確率測度の表式]
時刻 $n$ に位置 $x$ にいる確率は
\begin{equation}
  P(\tilde{S}_n = x) = \|\Xi_n(l,m)\,\varphi\|_1,
  \qquad l = \tfrac{n-x}{2},\; m = \tfrac{n+x}{2}
  \label{eq:crw_Xi}
\end{equation}
で与えられ，$\displaystyle\sum_{x} P(\tilde{S}_n = x) = 1$ を満たす。
\end{proposition}

\subsubsection*{時間発展対称な場合の具体計算}

$p = q$（時間発展対称）のとき，$a = d = p$，$b = c = 1 - p$ となる。
$q = 1 - p$ と置けば $p + q = 1$ であり，
\[
  A = \begin{pmatrix} p & q \\ q & p \end{pmatrix}, \quad
  P = \begin{pmatrix} p & q \\ 0 & 0 \end{pmatrix}, \quad
  Q = \begin{pmatrix} 0 & 0 \\ q & p \end{pmatrix}.
\]

\medskip
\noindent\textbf{$n=1$ のとき}（行列計算の詳細）：
\begin{align*}
  \widetilde{P}(\tilde{S}_1 = -1)
    &= P\varphi
     = \begin{pmatrix} p & q \\ 0 & 0 \end{pmatrix}
       \begin{pmatrix} \alpha \\ \beta \end{pmatrix}
     = \begin{pmatrix} p\alpha + q\beta \\ 0 \end{pmatrix},\\[4pt]
  \widetilde{P}(\tilde{S}_1 = +1)
    &= Q\varphi
     = \begin{pmatrix} 0 & 0 \\ q & p \end{pmatrix}
       \begin{pmatrix} \alpha \\ \beta \end{pmatrix}
     = \begin{pmatrix} 0 \\ q\alpha + p\beta \end{pmatrix}.
\end{align*}
よって
\[
  P(\tilde{S}_1 = -1) = p\alpha + q\beta, \qquad P(\tilde{S}_1 = +1) = q\alpha + p\beta.
\]
和は $(p + q)(\alpha + \beta) = 1$ となり，確率測度であることが確認できる。

\medskip
\noindent\textbf{$n=2$ のとき：}
\begin{align*}
  P(\tilde{S}_2 = -2) &= p\,(p\alpha + q\beta), \\
  P(\tilde{S}_2 =  0) &= q, \\
  P(\tilde{S}_2 = +2) &= p\,(q\alpha + p\beta).
\end{align*}
$P(\tilde{S}_2 = 0) = q$ は $\alpha + \beta = 1$，$p + q = 1$ から導かれる。

\medskip
\noindent\textbf{$n=3$ のとき：}
\begin{align*}
  P(\tilde{S}_3 = -3) &= p^2\,(p\alpha + q\beta), \\
  P(\tilde{S}_3 = -1) &= q^2\,(p\alpha + q\beta) + pq, \\
  P(\tilde{S}_3 = +1) &= q^2\,(q\alpha + p\beta) + pq, \\
  P(\tilde{S}_3 = +3) &= p^2\,(q\alpha + p\beta).
\end{align*}

\begin{remark}[偶奇性]
$n$ ステップ後の位置 $x$ は $n$ と同じ偶奇を持つ（$x \equiv n \pmod{2}$）。
これは通常のランダムウォークと同様である。
\end{remark}

\subsubsection*{平均}

\begin{proposition}[1次モーメント（時間発展対称な場合）]
$\Psi_0(0) = \varphi = (\alpha,\,\beta)^\top$，$p + q = 1$ とするとき，
\begin{equation}
  \mathbb{E}[\tilde{S}_n]
  = (\beta - \alpha)\sum_{k=1}^{n}(p - q)^k
  = \frac{(p-q)(\beta-\alpha)\bigl[1-(p-q)^n\bigr]}{2q}
  \qquad (p \neq q). \label{eq:crw_mean}
\end{equation}
$p = q = 1/2$ のとき $\mathbb{E}[\tilde{S}_n] = 0$，$\mathrm{Var}(\tilde{S}_n) = n$。
\end{proposition}

\begin{remark}
$|p - q| < 1$ であるから，$n \to \infty$ で $(p-q)^n \to 0$。
よって $\mathbb{E}[\tilde{S}_n]$ は有限値 $\dfrac{(p-q)(\beta-\alpha)}{2q}$
に収束する（$p \neq q$）。
これは通常のランダムウォークが $\mathbb{E}[S_n] \propto n$ で発散するのとは対照的である。
\end{remark}
